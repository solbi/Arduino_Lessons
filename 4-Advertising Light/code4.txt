int BASE = 2 ;  //第一w LED 接的 I/O _
int NUM = 6;   //LED 的

void setup()
{
   for (int i = BASE; i < BASE + NUM; i ++) 
   {
     pinMode(i, OUTPUT);   //O定底I/O_檩出
   }
}

void loop()
{
   for (int i = BASE; i < BASE + NUM; i ++) 
   {
     digitalWrite(i, LOW);    //O定底I/O_出"低"，即逐uP
     delay(200);        //延t
   }
   for (int i = BASE; i < BASE + NUM; i ++) 
   {
     digitalWrite(i, HIGH);    //O定底I/O_出"低"，即逐u_
     delay(200);        //延t
   }  
}
